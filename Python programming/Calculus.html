計算加總


import argparse

parser = argparse.ArgumentParser()
parser.add_argument("-s", "--sum", nargs="+", type=int
                    , required=True,help="計算加總")
args = parser.parse_args() #解析命令參數
total = 0

print("參數列表：", args.sum)
for val in args.sum:
    total += val

print("加總結果：", total)

input("Press Enter to exit...")

weight = float(input("請輸入您的體重(公斤)\n"))
BMI = round(weight/(height/100)**2,1)

if BMI < 18.5:
  print(f"您的BMI是：{BMI} (體重過輕)")
elif BMI < 24:
  print(f"您的BMI是：{BMI} (正常體位)")
elif BMI < 27:
  print(f"您的BMI是：{BMI} (體重過重)")
elif BMI < 30:
  print(f"您的BMI是：{BMI} (輕度肥胖)")
elif BMI < 35:
  print(f"您的BMI是：{BMI} (中度肥胖)")
else:
  print(f"您的BMI是：{BMI} (重度肥胖)")

綜合健康計算機 Comprehensive Health Computer


# 綜合健康計算機 Comprehensive Health Computer
# 計算BMI、BMR(基礎代謝率)、TDEE(總熱量消耗)

def get_bmi(height, weight):
    bmi = weight/(height/100)**2
    return round(bmi, 1)

def get_bmr(gender, height, weight, age):
    if gender == "男":
        bmr = 66 + 13.7 * weight + 5 * height - 6.8 * age
    else:
        bmr = 655 + 9.6 * weight + 1.8 * height - 4.7 * age
    return round(bmr, 2)

def get_tdee(gender, height, weight, age, act):
    bmr = get_bmr(gender, height, weight, age)
    if act == "1":
        tdee = bmr * 1.2
    elif act == "2":
        tdee = bmr * 1.375
    elif act == "3":
        tdee = bmr * 1.55
    elif act == "4":
        tdee = bmr * 1.725
    elif act == "5":
        tdee = bmr * 1.9
    return round(tdee, 2)

print("歡迎使用綜合健康計算機~")
print("(1)計算BMI")
print("(2)計算基礎代謝率(BMR)")
print("(2)計算總熱量消耗(TDEE)")
item = input("請選擇要計算的項目 (輸入1 or 2 or 3)：")
if item == "1":
    height = float(input("請輸入您的身高(公分)："))
    weight = float(input("請輸入您的體重(公斤)："))
    bmi = get_bmi(height, weight)
    print(f"您的BMI：{bmi}")
elif item == "2":
    gender = str(input("請輸入您的性別(男or女)："))
    height = float(input("請輸入您的身高(公分)："))
    weight = float(input("請輸入您的體重(公斤)："))
    age = int(input("請輸入您的年齡："))
    bmr = get_bmr(gender, height, weight, age)
    print(f"您的基礎代謝率(BMR)：{bmr}")
else:
    gender = str(input("請輸入您的性別(男or女)："))
    height = float(input("請輸入您的身高(公分)："))
    weight = float(input("請輸入您的體重(公斤)："))
    age = int(input("請輸入您的年齡："))
    print("(1)久坐、幾乎沒運動")
    print("(2)每周低強度運動1~3天")
    print("(3)每周中強度運動3~5天")
    print("(4)每周高強度運動6~7天")
    print("(5)勞力密集工作或是每天高強度運動")
    act = input("請選擇您的運動量 (輸入1~5)：")
    tdee = get_tdee(gender, height, weight, age, act)
    print(f"您的總熱量消耗(TDEE)：{tdee}")

input("Press Enter to exit...")
